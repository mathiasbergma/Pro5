108int initModule(int timeout)
109{
110 LTEShieldSerial.begin(9600);
111 unsigned long startTime = millis();
112
113 printToConsole("Transmitting AT\n");
114 while (!getResponse("OK", 500))
115 {
116 printToConsole(".");
117 transmitCommand("AT"); // Send AT command to enable the interface
118
119 if (millis() - startTime > timeout)
120 {
121 return 0;
122 }
123 }
124 printToConsole("Got OK\n");
125
126 printToConsole("Disabling echo\n");
127 transmitCommand("ATE0"); // Disable echo
128 if (getResponse("OK", 200))
129 {
130 printToConsole("Echo disabled");
131 }
132 else
133 {
134 printToConsole("Echo not disabled");
135 }
136
137 transmitCommand("AT+CTZU=1"); // Enable automatic time zone update
138 if (getResponse("OK", 5000))
139 {
140 printToConsole("Automatic time zone update enabled");
141 }
142 else
143 {
144 printToConsole("Automatic time zone update not enabled");
145 }
146 return 1;
